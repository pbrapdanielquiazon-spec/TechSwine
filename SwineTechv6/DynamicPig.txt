<script>
  function showView(view) {
    document.querySelectorAll(".record-view").forEach(v => v.classList.remove("active"));
    document.getElementById(view).classList.add("active");
    document.querySelectorAll(".tab-btn").forEach(b => b.classList.remove("active"));
    document.querySelector(`.tab-btn[onclick="showView('${view}')"]`).classList.add("active");
  }
  function openModal(id){ document.getElementById(id).style.display = "block"; }
  function closeModal(id){ document.getElementById(id).style.display = "none"; }

  // Dynamic Add Functions
  function addLitter(e){
    e.preventDefault();
    let sow = addSowId.value, birth = addBirthDate.value, size = addLitterSize.value, caretaker = addCaretakerId.value;
    if(!sow||!birth||size<=0||!caretaker){ openModal("errorModal"); return false; }
    litterTable.insertRow().innerHTML = `<td>NEW</td><td>${sow}</td><td>${birth}</td><td>${size}</td><td>${caretaker}</td>
      <td><button class='btn-edit'>Edit</button><button class='btn-archive'>Archive</button></td>`;
    closeModal("addLitterModal"); openModal("successModal"); return true;
  }
  function addPig(e){
    e.preventDefault();
    let litter = pigLitterId.value, sow = sowIdentifier.value, birth = pigBirthDate.value, status = pigStatus.value, avail = pigAvailability.value, notes = pigNotes.value;
    if(!litter||!sow||!birth){ openModal("errorModal"); return false; }
    pigTable.insertRow().innerHTML = `<td>NEW</td><td>${litter}</td><td>${sow}</td><td>${birth}</td><td>${status}</td><td>${avail}</td><td>${notes}</td>
      <td><button class='btn-edit'>Edit</button><button class='btn-archive'>Archive</button></td>`;
    closeModal("addPigModal"); openModal("successModal"); return true;
  }
  function addHealth(e){
    e.preventDefault();
    let pigId = healthPigId.value, sym = symptoms.value, diag = diagnosis.value, treat = treatment.value, mort = mortality.value, date = recordedAt.value, caretaker = healthCaretaker.value;
    if(!pigId||!sym||!diag||!date){ openModal("errorModal"); return false; }
    healthTable.insertRow().innerHTML = `<td>NEW</td><td>${pigId}</td><td>${sym}</td><td>${diag}</td><td>${treat}</td><td>${mort}</td><td>${date}</td><td>${caretaker}</td>
      <td><button class='btn-edit'>Edit</button><button class='btn-archive'>Archive</button></td>`;
    closeModal("addHealthModal"); openModal("successModal"); return true;
  }
  function addBreeding(e){
    e.preventDefault();
    let sow = breedingSow.value, boar = breedingBoar.value, date = breedingDate.value, farrow = expectedFarrowing.value, outcome = breedingOutcome.value;
    if(!sow||!boar||!date||!farrow){ openModal("errorModal"); return false; }
    breedingTable.insertRow().innerHTML = `<td>NEW</td><td>${sow}</td><td>${boar}</td><td>${date}</td><td>${farrow}</td><td>${outcome}</td>
      <td><button class='btn-edit'>Edit</button><button class='btn-archive'>Archive</button></td>`;
    closeModal("addBreedingModal"); openModal("successModal"); return true;
  }
</script>
